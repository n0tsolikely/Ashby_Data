import logging
from typing import List, Union

from tuya_iot import TuyaOpenAPI, TUYA_LOGGER
from env import ENDPOINT, ACCESS_ID, ACCESS_KEY, USERNAME, PASSWORD
from ashpi_devices import DEVICES

# ---------- Logging ----------
logging.basicConfig(
    format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
    level=logging.INFO,
)
TUYA_LOGGER.setLevel(logging.INFO)
logger = logging.getLogger("ashpi_lights_tuya")

# ---------- Tuya client ----------
openapi = TuyaOpenAPI(ENDPOINT, ACCESS_ID, ACCESS_KEY)
openapi.connect(USERNAME, PASSWORD, "1", "smartlife")

# ---------- Local brightness cache ----------
# name -> 0–1000
BRIGHTNESS_STATE = {}

DEFAULT_STEP = 100
DEFAULT_BRIGHTNESS = 500  # used if we don't know current brightness yet


def _normalize_names(target: Union[str, List[str]]) -> List[str]:
    if isinstance(target, str):
        return [target]
    return list(target)


def _ensure_bulb(name: str):
    if name not in DEVICES:
        raise ValueError(f"Unknown device: {name}")

    # ✅ NEW: use the 'category' field from ashpi_devices.py
    dev_category = DEVICES[name].get("category")
    allowed_bulb_categories = {"bulb", "light"}

    if dev_category not in allowed_bulb_categories:
        raise ValueError(
            f"Device {name} is not a bulb (category={dev_category})"
        )


def _get_bulb_ids(target: Union[str, List[str]]):
    names = _normalize_names(target)
    bulbs = []
    for n in names:
        _ensure_bulb(n)
        bulbs.append((n, DEVICES[n]["id"]))
    return bulbs


def _clamp(value: int, min_v: int = 0, max_v: int = 1000) -> int:
    return max(min_v, min(max_v, int(value)))


# ---------- Core actions ----------

def set_brightness(target: Union[str, List[str]], value: int) -> int:
    """
    Set brightness (0–1000) for one or many Tuya bulbs.
    Also turns the LED ON if value > 0, OFF if value == 0.
    """
    bulbs = _get_bulb_ids(target)
    value = _clamp(value)

    for name, dev_id in bulbs:
        BRIGHTNESS_STATE[name] = value

        commands = {
            "commands": [
                {"code": "bright_value_v2", "value": value},
                {"code": "switch_led", "value": value > 0},
            ]
        }
        res = openapi.post(f"/v1.0/devices/{dev_id}/commands", commands)
        logger.info("set_brightness %s -> %s | %s", name, value, res)

    return value


def step_brightness(target: Union[str, List[str]], delta: int = DEFAULT_STEP) -> None:
    """
    Increase or decrease brightness by delta for one or many bulbs.
    Positive delta = brighter, negative delta = dimmer.
    """
    bulbs = _get_bulb_ids(target)

    for name, _ in bulbs:
        current = BRIGHTNESS_STATE.get(name, DEFAULT_BRIGHTNESS)
        new_value = _clamp(current + delta)
        set_brightness(name, new_value)


def turn_on(target: Union[str, List[str]], value: int | None = None) -> None:
    """
    Turn bulb(s) on. If value is None, use cached brightness or DEFAULT_BRIGHTNESS.
    """
    bulbs = _get_bulb_ids(target)

    # If a single bulb and value is None, use its own cache
    if value is None and isinstance(target, str):
        value = BRIGHTNESS_STATE.get(target, DEFAULT_BRIGHTNESS)

    # If still None (e.g. list), fall back to default
    if value is None:
        value = DEFAULT_BRIGHTNESS

    set_brightness(target, value)


def turn_off(target: Union[str, List[str]]) -> None:
    """
    Turn bulb(s) off (switch_led = False, brightness cached as 0).
    """
    bulbs = _get_bulb_ids(target)

    for name, dev_id in bulbs:
        BRIGHTNESS_STATE[name] = 0
        commands = {
            "commands": [
                {"code": "switch_led", "value": False},
            ]
        }
        res = openapi.post(f"/v1.0/devices/{dev_id}/commands", commands)
        logger.info("turn_off %s | %s", name, res)


def get_cached_brightness(name: str) -> int:
    """
    Return last cached brightness for a bulb, or DEFAULT_BRIGHTNESS if unknown.
    (We can later upgrade this to query Tuya live if we want.)
    """
    _ensure_bulb(name)
    return BRIGHTNESS_STATE.get(name, DEFAULT_BRIGHTNESS)

from ashpi_lights_tuya import set_brightness as tuya_set_brightness, turn_off as tuya_turn_off

# --- Logical group → physical devices mapping --- #
# group name (what NLU/router uses) → list of device keys from DEVICES
GROUPS = {
    "captain_america": ["captain_america_bulb"],
    "thor": ["thor_bulb"],
    "sky": ["sky_bulb"],
    # later:
    # "bed_lamp": ["bed_lamp"],
}


def set_group_brightness(group: str, level: int):
    """
    group: e.g. 'captain_america'
    level: 0–1000 internal scale

    For bulbs:
      - level > 0  → set brightness via Tuya
      - level == 0 → turn off

    For plugs / other categories:
      - currently just logs (we can wire Tuya plugs later).
    """

    device_keys = GROUPS.get(group)
    if not device_keys:
        print(f"[AshPi WARNING] No devices configured for group '{group}'")
        return

    for key in device_keys:
        info = DEVICES.get(key)
        if not info:
            print(f"[AshPi WARNING] Device key '{key}' missing in DEVICES")
            continue

        category = info.get("category")

        # Bulbs -> Tuya brightness control
        if category == "bulb":
            if level <= 0:
                tuya_turn_off(key)
            else:
                tuya_set_brightness(key, level)
            continue

        # Plugs (future)
        if category == "plug":
            # TODO: implement real Tuya plug on/off.
            on = level > 0
            print(
                f"[AshPi TODO] Would set plug '{key}' "
                f"(id={info.get('id')}) to {'ON' if on else 'OFF'}"
            )
            continue

        # Fallback for unknown categories
        print(
            f"[AshPi WARNING] Unknown category '{category}' "
            f"for device '{key}' in group '{group}'"
        )

import re

def parse_lighting(text: str):
    t = text.lower().strip()

    # 1) "kill it" → turn off last active group
    if "kill it" in t:
        return {
            "type": "lights.off",
            "relative": False,
            "direction": "down",
        }

    # 2) "30% less / 20% more"
    m = re.search(r"(\d+)\s*%?\s*(less|more)", t)
    if m:
        pct = int(m.group(1))
        direction = "down" if m.group(2) == "less" else "up"
        return {
            "type": "lights.change",
            "mode": "relative_percent",
            "percent": pct,
            "direction": direction,
        }

    # 3) plain "30%" (no 'less/more') → absolute level
    m2 = re.search(r"\b(\d+)\s*%\b", t)
    if m2:
        pct = int(m2.group(1))
        return {
            "type": "lights.set",
            "mode": "absolute_percent",
            "percent": pct,
        }

    # ... fall through to your existing logic for "more / less / dim it / brighter"
    return None
