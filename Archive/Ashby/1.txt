# ashpi/brain/lighting_sessions.py
import time
from dataclasses import dataclass, field
from typing import Optional, Dict, Any

SESSION_TTL = 30.0  # seconds


@dataclass
class LightingSession:
    last_group: Optional[str] = None
    last_direction: Optional[str] = None  # "up" / "down"
    awaiting_target_for: Optional[Dict[str, Any]] = None  # pending intent (e.g. power=on)
    last_updated: float = field(default_factory=time.time)


_sessions: Dict[str, LightingSession] = {}


def _fresh_session() -> LightingSession:
    return LightingSession()


def get_session(user_id: str) -> LightingSession:
    """Get or create a session. If too old, reset."""
    now = time.time()
    session = _sessions.get(user_id)

    if session is None or now - session.last_updated > SESSION_TTL:
        session = _fresh_session()
        _sessions[user_id] = session

    session.last_updated = now
    return session


def set_last_group(user_id: str, group: str) -> None:
    session = get_session(user_id)
    session.last_group = group


def set_last_direction(user_id: str, direction: Optional[str]) -> None:
    session = get_session(user_id)
    session.last_direction = direction


def set_awaiting_target(user_id: str, pending_intent: Dict[str, Any]) -> None:
    """Store an intent that needs a target (e.g. user said 'turn it on')."""
    session = get_session(user_id)
    session.awaiting_target_for = pending_intent


def pop_awaiting_target(user_id: str) -> Optional[Dict[str, Any]]:
    """Return and clear pending target intent."""
    session = get_session(user_id)
    pending = session.awaiting_target_for
    session.awaiting_target_for = None
    return pending
