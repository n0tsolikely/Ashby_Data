import time

from ashpi.control.device_control import set_group_brightness

from . import comfort_sessions, memory, lighting_sessions
from .nlu_manager import extract_intent

_brains: Dict[str, AshBrain] = {}
_light_sessions: Dict[str, Dict[str, object]] = {}
_recent_actions: Dict[str, Dict[str, Any]] = {}

def get_brain(user_id: str) -> AshBrain:
    user_id = str(user_id)
    if user_id not in _brains:
        _brains[user_id] = AshBrain(user_id)
    return _brains[user_id]

def _get_light_session(user_id: str):
    return _light_sessions.get(str(user_id))

def _start_light_session(user_id: str, group: str, level: int, last_delta: int = 0):
    _light_sessions[str(user_id)] = {"group": group, "level": level, "last_delta": int(last_delta)}

def _update_light_level(user_id: str, level: int):
    if str(user_id) in _light_sessions:
        _light_sessions[str(user_id)]["level"] = level

def _end_light_session(user_id: str):
    _light_sessions.pop(str(user_id), None)

def _clamp_level(level: int) -> int:
    return max(0, min(1000, int(level)))

def _set_recent_action(user_id: str, action_type: str, **data: Any) -> None:
    _recent_actions[str(user_id)] = {"type": action_type, "ts": time.time(), **data}

def _get_recent_action(user_id: str, max_age: float = 30.0):
    rec = _recent_actions.get(str(user_id))
    if not rec:
        return None
    if time.time() - rec["ts"] > max_age:
        return None
    return rec

def handle_text(user_id: str, text: str) -> str:
    user_id = str(user_id)
    text_stripped = text.strip()
    brain = get_brain(user_id)

    if comfort_sessions.has_active_session(user_id):
        temp, msg = comfort_sessions.process_reply(user_id, text_stripped)
        return msg

    intent = extract_intent(text_stripped)
    pending = lighting_sessions.pop_awaiting_target(user_id)

    itype = intent["type"]
    group = intent.get("group")
    brightness = intent.get("brightness")
    delta = intent.get("delta")
    mode = intent.get("mode")

    if pending and pending.get("type") == "lights.on":
        if not group:
            low = text_stripped.lower().strip()
            for f in ["man", "bro", "dude"]:
                if low.endswith(" " + f):
                    low = low[: -(len(f) + 1)].strip()
            if low.startswith("the "):
                low = low[4:]
            if low not in ["turn it on", "turn it off", "on", "off"]:
                group = low.replace(" ", "_")

        if brightness is None and pending.get("brightness") is not None:
            brightness = pending["brightness"]
        if mode is None and pending.get("mode") is not None:
            mode = pending["mode"]

        if group and not itype.startswith("lights."):
            itype = "lights.on"

    # ACK logic
    low = text_stripped.lower()
    recent = _get_recent_action(user_id)
    if recent and recent["type"] == "lighting" and itype in ("chat", "unknown"):
        acks = ["perfect", "better", "nice", "good", "awesome", "sweet", "beauty", "thanks"]
        if any(a in low for a in acks):
            grp = recent["group"]
            level = recent["level"]
            nice = grp.replace("_", " ")
            percent = round(level / 10)
            try:
                memory.set(user_id, f"preferred_level_{grp}", level)
            except:
                pass
            return f"Gotcha. Leaving {nice} around {percent}%."

    # ON
    if itype == "lights.on":
        if not group:
            lighting_sessions.set_awaiting_target(user_id, intent)
            return "Which lights, man?"
        level = 600 if brightness is None else _clamp_level(brightness)
        set_group_brightness(group, level)
        _start_light_session(user_id, group, level, 0)
        _set_recent_action(user_id, "lighting", group=group, level=level)
        nice = group.replace("_", " ")
        return f"Turning on {nice}."

    # OFF
    if itype == "lights.off":
        sess = _get_light_session(user_id)
        target = group or (sess["group"] if sess else None)
        if not target:
            return brain.chat(text_stripped)
        set_group_brightness(target, 0)
        _end_light_session(user_id)
        _set_recent_action(user_id, "lighting", group=target, level=0)
        nice = target.replace("_", " ")
        return f"Turning off {nice}."

    # ADJUST
    if itype == "lights.adjust":
        sess = _get_light_session(user_id)
        if sess and not group:
            target = sess["group"]
        else:
            target = group
        if not target:
            return brain.chat(text_stripped)

        if sess and sess["group"] == target:
            level = int(sess["level"])
            last_delta = int(sess["last_delta"])
        else:
            level = 600
            last_delta = 0

        if brightness is not None:
            level = _clamp_level(brightness)
            last_delta = 0
        else:
            if delta is None and last_delta != 0:
                delta = last_delta
            if delta is not None:
                step = 100
                delta = int(delta)
                if abs(delta) < step:
                    delta = step if delta > 0 else -step
                level = _clamp_level(level + delta)
                last_delta = delta

        set_group_brightness(target, level)
        _start_light_session(user_id, target, level, last_delta)
        _set_recent_action(user_id, "lighting", group=target, level=level)
        nice = target.replace("_", " ")
        percent = round(level / 10)
        return f"Setting {nice} to {percent}%."

    return brain.chat(text_stripped)

base_router = r'''"""
AshPi Router
============

This module is the main text router for AshPi. It is responsible for:
- Receiving raw user text (from Telegram, CLI, HTTP, etc.)
- Passing it through NLU to extract an intent
- Routing the intent to the correct domain handler (lights, comfort, small talk, etc.)
- Maintaining short-lived "sessions" (e.g., adjusting the same light repeatedly)
- Logging and providing debug helpers

This file is intentionally verbose and heavily commented so it doubles as living
documentation for the AshPi brain. It is also deliberately kept as a single
module in this variant so we can test very large-file behavior in tools.
"""

import logging
import time
from dataclasses import dataclass, field
from typing import Dict, Any, Optional, List, Tuple, Callable

from ashpi.control.device_control import set_group_brightness
from .ash_core import AshBrain
from . import comfort_sessions, memory, lighting_sessions
from .nlu_manager import extract_intent

logger = logging.getLogger(__name__)
if not logger.handlers:
    logging.basicConfig(
        format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
        level=logging.INFO,
    )

# ---------------------------------------------------------------------------
# Data classes and in-memory registries
# ---------------------------------------------------------------------------

@dataclass
class LightSession:
    """Represents a short-lived lighting interaction window.

    When the user says things like:

        "turn on the bed lamp"
        "a bit darker"
        "more"
        "ok up a bit"

    we treat all of that as a single "session" associated with:
    - a specific user_id
    - a target group (e.g. "bed_lamp" group)
    - the last known absolute level (0–1000)
    - the last signed delta (e.g. -100, +200)

    The session is normally expired implicitly after a while, but in this
    simple in-memory variant we just let it be overwritten or dropped when
    the user issues something unrelated.
    """
    group: str
    level: int
    last_delta: int = 0
    started_at: float = field(default_factory=time.time)

@dataclass
class RecentAction:
    """A short-lived record of the last "important" action.

    For example:
    - last lighting group + level
    - last comfort adjustment
    - last scene trigger

    This allows AshPi to react meaningfully to "nice", "perfect", "better",
    etc., by remembering what just happened and confirming that state.
    """
    type: str
    ts: float
    data: Dict[str, Any] = field(default_factory=dict)


# In long-running processes (like a Telegram bot) we keep brains in memory:
_brains: Dict[str, AshBrain] = {}

# Simple in-memory lighting sessions:
_light_sessions: Dict[str, LightSession] = {}

# Short-lived recent actions (one per user):
_recent_actions: Dict[str, RecentAction] = {}

# ---------------------------------------------------------------------------
# Utility helpers
# ---------------------------------------------------------------------------

def get_brain(user_id: str) -> AshBrain:
    """Return an AshBrain instance for a given user_id, creating it if needed."""
    user_id = str(user_id)
    if user_id not in _brains:
        logger.info("Creating AshBrain for user_id=%s", user_id)
        _brains[user_id] = AshBrain(user_id)
    return _brains[user_id]


def _get_light_session(user_id: str) -> Optional[LightSession]:
    """Return the LightSession for a user, if any."""
    return _light_sessions.get(str(user_id))


def _start_light_session(user_id: str, group: str, level: int, last_delta: int = 0) -> None:
    """Begin or replace the light session for a user."""
    sess = LightSession(group=group, level=int(level), last_delta=int(last_delta))
    _light_sessions[str(user_id)] = sess
    logger.debug("Started LightSession for user=%s group=%s level=%s delta=%s",
                 user_id, group, level, last_delta)


def _update_light_level(user_id: str, level: int) -> None:
    """Update only the level of an existing light session, if present."""
    key = str(user_id)
    if key in _light_sessions:
        _light_sessions[key].level = int(level)
        logger.debug("Updated LightSession level for user=%s to %s", user_id, level)


def _end_light_session(user_id: str) -> None:
    """End (delete) the light session for a user, if any."""
    popped = _light_sessions.pop(str(user_id), None)
    if popped:
        logger.debug("Ended LightSession for user=%s (group=%s)", user_id, popped.group)


def _clamp_level(level: int) -> int:
    """Clamp a brightness level to the valid 0–1000 range."""
    return max(0, min(1000, int(level)))


def _set_recent_action(user_id: str, action_type: str, **data: Any) -> None:
    """Record the last "important" action for a user.

    This is intentionally tiny, but gives us a hook for follow-up acknowledgements.
    """
    _recent_actions[str(user_id)] = RecentAction(
        type=action_type,
        ts=time.time(),
        data=data or {},
    )
    logger.debug("Set RecentAction for user=%s type=%s data=%r", user_id, action_type, data)


def _get_recent_action(user_id: str, max_age: float = 30.0) -> Optional[RecentAction]:
    """Return the recent action if it is still fresh enough, else None."""
    ra = _recent_actions.get(str(user_id))
    if not ra:
        return None
    if time.time() - ra.ts > max_age:
        return None
    return ra


def _format_group_name(group: str) -> str:
    """Return a human-nice version of a group name."""
    return group.replace("_", " ")


def _percent_from_level(level: int) -> int:
    """Convert internal 0–1000 level to rough percent for chat."""
    return round(int(level) / 10)


def _normalize_followup_raw_text(text: str) -> str:
    """Normalize short follow-up replies like 'bed lamp bro', 'the lamp', etc.

    This is used when a pending lighting intent is waiting for the user to
    specify *which* lights they meant.
    """
    low = text.lower().strip()
    # Strip trailing filler words that humans love throwing in:
    for filler in ["man", "bro", "dude", "please", "thanks", "thank you"]:
        if low.endswith(" " + filler):
            low = low[: -(len(filler) + 1)].strip()

    # Strip leading "the " when referring to a group by friendly name:
    if low.startswith("the "):
        low = low[4:]

    return low


def _resolve_pending_group_name(pending_intent: Dict[str, Any], text_stripped: str) -> Optional[str]:
    """Try to infer a light group from a follow-up free-form reply.

    Example:
        User: "turn on the lights"
        AshPi: "Which lights, man?"
        User: "the bed lamp"

    The pending intent is a lights.on with no group; this function interprets
    "the bed lamp" into "bed_lamp".
    """
    low = _normalize_followup_raw_text(text_stripped)
    if low in ["turn it on", "turn it off", "on", "off"]:
        # These don't contain a concrete group
        return None
    if not low:
        return None
    return low.replace(" ", "_")

def _maybe_handle_acknowledgement(user_id: str, text_stripped: str, itype: str) -> Optional[str]:
    """Handle 'perfect', 'better', 'nice', etc. after a recent action.

    If this returns a string, the caller should immediately return it to the user.
    """
    low = text_stripped.lower()
    recent = _get_recent_action(user_id)

    if not recent:
        return None
    if recent.type != "lighting":
        return None
    if itype not in ("chat", "unknown"):
        return None

    acks = [
        "perfect",
        "better",
        "nice",
        "good",
        "awesome",
        "sweet",
        "beauty",
        "thanks",
        "thank you",
        "that works",
        "love it",
        "leave it",
    ]
    if not any(a in low for a in acks):
        return None

    grp = recent.data.get("group")
    level = recent.data.get("level")
    if not grp or level is None:
        return None

    nice = _format_group_name(grp)
    percent = _percent_from_level(level)

    # Try to store a "preferred" brightness for this group in memory
    try:
        memory.set(user_id, f"preferred_level_{grp}", level)
    except Exception as e:  # We really don't care if this fails
        logger.debug("Failed to store preferred level for %s: %s", grp, e)

    return f"Gotcha. Leaving {nice} around {percent}%."


def _handle_lights_on(
    user_id: str,
    text_stripped: str,
    intent: Dict[str, Any],
    pending: Optional[Dict[str, Any]],
) -> str:
    """Handle lights.on intent with optional pending follow-up resolution."""
    itype = intent["type"]
    group = intent.get("group")
    brightness = intent.get("brightness")
    mode = intent.get("mode")

    # If we were waiting for the user to specify a target light group:
    if pending and pending.get("type") == "lights.on":
        if not group:
            group = _resolve_pending_group_name(pending, text_stripped)

        # Inherit brightness/mode from pending if not overridden:
        if brightness is None and pending.get("brightness") is not None:
            brightness = pending["brightness"]
        if mode is None and pending.get("mode") is not None:
            mode = pending["mode"]

        if group and not itype.startswith("lights."):
            itype = "lights.on"  # Force back to lighting domain

    if not group:
        # We still don't know which lights — ask and store the partial intent:
        lighting_sessions.set_awaiting_target(user_id, intent)
        return "Which lights, man?"

    level = 600 if brightness is None else _clamp_level(brightness)
    set_group_brightness(group, level)
    _start_light_session(user_id, group, level, 0)
    _set_recent_action(user_id, "lighting", group=group, level=level)
    nice = _format_group_name(group)
    return f"Turning on {nice}."


def _handle_lights_off(user_id: str, text_stripped: str, intent: Dict[str, Any]) -> str:
    """Handle lights.off intent, falling back to chat if nothing is targeted."""
    brain = get_brain(user_id)
    group = intent.get("group")

    sess = _get_light_session(user_id)
    target = group or (sess.group if sess else None)
    if not target:
        return brain.chat(text_stripped)

    set_group_brightness(target, 0)
    _end_light_session(user_id)
    _set_recent_action(user_id, "lighting", group=target, level=0)
    nice = _format_group_name(target)
    return f"Turning off {nice}."


def _handle_lights_adjust(user_id: str, text_stripped: str, intent: Dict[str, Any]) -> str:
    """Handle lights.adjust intent for brightness tweaks and absolute levels."""
    brain = get_brain(user_id)
    group = intent.get("group")
    brightness = intent.get("brightness")
    delta = intent.get("delta")

    sess = _get_light_session(user_id)
    if sess and not group:
        target = sess.group
    else:
        target = group

    if not target:
        return brain.chat(text_stripped)

    if sess and sess.group == target:
        level = int(sess.level)
        last_delta = int(sess.last_delta)
    else:
        # If we don't have a session, assume a default:
        level = 600
        last_delta = 0

    if brightness is not None:
        # Absolute brightness takes precedence over delta adjustments
        level = _clamp_level(brightness)
        last_delta = 0
    else:
        # No explicit absolute level; apply delta or inherit last direction
        if delta is None and last_delta != 0:
            delta = last_delta
        if delta is not None:
            step = 100
            delta = int(delta)
            if abs(delta) < step:
                delta = step if delta > 0 else -step
            level = _clamp_level(level + delta)
            last_delta = delta

    set_group_brightness(target, level)
    _start_light_session(user_id, target, level, last_delta)
    _set_recent_action(user_id, "lighting", group=target, level=level)
    nice = _format_group_name(target)
    percent = _percent_from_level(level)
    return f"Setting {nice} to {percent}%."


def handle_text(user_id: str, text: str) -> str:
    """Main entry point.

    This is the only function that external callers should use. It receives
    raw user text and returns a chat-style string reply.
    """
    user_id = str(user_id)
    text_stripped = text.strip()
    brain = get_brain(user_id)

    # 1) Active "comfort" session short-circuits everything else:
    if comfort_sessions.has_active_session(user_id):
        # For comfort, we only care about the textual reply; the temperature
        # or numeric side is handled by the comfort module.
        _temp, msg = comfort_sessions.process_reply(user_id, text_stripped)
        return msg

    # 2) NLU: extract intent from the raw text
    intent = extract_intent(text_stripped)

    # 3) If we had a pending lighting target, consume it here
    pending = lighting_sessions.pop_awaiting_target(user_id)

    itype = intent["type"]

    # 4) Special acknowledgement hook — "perfect", "better", etc.
    ack = _maybe_handle_acknowledgement(user_id, text_stripped, itype)
    if ack is not None:
        return ack

    # 5) Domain routing
    if itype == "lights.on":
        return _handle_lights_on(user_id, text_stripped, intent, pending)
    if itype == "lights.off":
        return _handle_lights_off(user_id, text_stripped, intent)
    if itype == "lights.adjust":
        return _handle_lights_adjust(user_id, text_stripped, intent)

    # 6) Fallback to general chat
    return brain.chat(text_stripped)

def _debug_dump_sessions() -> Dict[str, Any]:
    """Return a debug snapshot of all in-memory sessions and recent actions.

    This is not meant for end users; it's primarily for logging, tests, or
    introspection via an internal admin channel.
    """
    lights = {
        user_id: {
            "group": sess.group,
            "level": sess.level,
            "last_delta": sess.last_delta,
            "started_at": sess.started_at,
        }
        for user_id, sess in _light_sessions.items()
    }

    recents = {
        user_id: {
            "type": ra.type,
            "ts": ra.ts,
            "data": dict(ra.data),
        }
        for user_id, ra in _recent_actions.items()
    }

    return {
        "light_sessions": lights,
        "recent_actions": recents,
    }


def _debug_pretty_state() -> str:
    """Return a multi-line human-readable dump of sessions and recents."""
    snap = _debug_dump_sessions()
    lines: List[str] = []
    lines.append("=== AshPi Router State ===")
    lines.append("")
    lines.append("Light sessions:")
    if not snap["light_sessions"]:
        lines.append("  (none)")
    else:
        for uid, info in snap["light_sessions"].items():
            lines.append(
                f"  user={uid} group={info['group']} level={info['level']} "
                f"delta={info['last_delta']} started_at={info['started_at']}"
            )

    lines.append("")
    lines.append("Recent actions:")
    if not snap["recent_actions"]:
        lines.append("  (none)")
    else:
        for uid, info in snap["recent_actions"].items():
            lines.append(
                f"  user={uid} type={info['type']} ts={info['ts']} data={info['data']}"
            )

    return "\n".join(lines)

AshPi Router Design Notes
-------------------------

This section documents the design philosophy behind this router. It is also
purposely long so we can stress-test large source files in various tools.

1. Single Entry Point
   The handle_text(user_id, text) function is the only public entry point.
   All callers send raw user text here, regardless of transport (Telegram,
   HTTP, CLI, test harness, etc.). This keeps external integration dirt-simple.

2. Domain Separation
   The router itself knows about domains *only at the boundary layer*:
   - "lights" (on/off/adjust)
   - "comfort" (temperature intent sessions)
   - "chat" (general conversation through AshBrain)

   The actual business logic for devices, temperature calculation, etc. lives
   in separate modules (device_control, comfort_sessions, etc.). The router's
   job is to interpret user intent and pick which domain should respond.

3. Memory and Preferences
   When a user acknowledges a lighting change as "perfect" or "better", we
   opportunistically write that brightness level into the memory module as
   "preferred_level_<group>". The router doesn't care *how* memory is stored;
   it only treats memory as a key/value service owned by the core brain.

4. Session Handling
   Short-lived sessions (LightSession) make follow-up phrasing natural:
   - "turn on the bed lamp"
   - "darker"
   - "more"
   - "okay a bit brighter"

   Without sessions, each utterance would have to re-specify the target
   group and sometimes the direction, making the UX clunky.

5. Extensibility
   In a larger AshPi deployment, this router could grow a pluggable registry
   of "intent handlers" keyed by domain + intent type. For now, we hard-code
   lights.* intents because lighting is the first major device domain.

6. Debugging
   The *_debug_* helpers are intentionally verbose and string-based so they
   can be dumped directly into logs or into a developer-only admin chat.
   They are not meant to be part of the public smart home experience.

7. Future Ideas
   - Scene management (e.g. "movie time", "bedtime", "gaming mode")
   - Multi-room disambiguation ("which bedroom?", "front or back?")
   - User-specific phrasing quirks, learned and stored via memory
   - Richer acknowledgement categories beyond lighting
   - Pluggable guardrails for safety / constraints

This block of notes is intentionally padded with detail so that this file
remains large enough to be a robust test of long-source-handling behavior.
"""

# ashpi_brain/nlu/local_nlu.py

import os
import difflib
from typing import Dict, List, Tuple, Optional

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
DATA_FILE = os.path.join(BASE_DIR, "data", "nlu_behemoth_examples.txt")


class LocalNLU:
    """
    Very simple local NLU:
    - Loads examples from a TSV file: intent<TAB>utterance
    - Uses fuzzy matching (difflib) to find the closest utterance
    - Returns (intent, score) for the best match
    """

    def __init__(self, data_file: str = DATA_FILE):
        self.data_file = data_file
        self.examples: Dict[str, List[str]] = {}
        self._loaded = False

    def load(self) -> None:
        if self._loaded:
            return
        if not os.path.exists(self.data_file):
            raise FileNotFoundError(f"LocalNLU data file not found: {self.data_file}")

        examples: Dict[str, List[str]] = {}
        with open(self.data_file, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line or line.startswith("#"):
                    continue
                if "\t" not in line:
                    continue
                intent, utterance = line.split("\t", 1)
                intent = intent.strip()
                utt = utterance.strip().lower()
                examples.setdefault(intent, []).append(utt)

        self.examples = examples
        self._loaded = True

    def classify(self, text: str) -> Tuple[Optional[str], float]:
        """
        Return (best_intent, score) for the given text.
        Score is 0.0–1.0 from difflib.SequenceMatcher.
        """
        self.load()
        text_norm = text.strip().lower()
        if not text_norm:
            return None, 0.0

        best_intent: Optional[str] = None
        best_score: float = 0.0

        for intent, utterances in self.examples.items():
            for utt in utterances:
                score = difflib.SequenceMatcher(a=text_norm, b=utt).ratio()
                if score > best_score:
                    best_score = score
                    best_intent = intent

        return best_intent, best_score


# Convenience singleton
_local_nlu = LocalNLU()


def classify(text: str) -> Tuple[Optional[str], float]:
    """
    Module-level helper so other code can just call local_nlu.classify(text).
    """
    return _local_nlu.classify(text)

from .nlu.local_nlu import classify as local_classify

def _default_intent() -> Dict[str, Any]:
    return {
        "type": "unknown",
        "group": None,
        "brightness": None,
        "delta": None,
        "temperature": None,
    }


def _map_local_intent_to_schema(local_intent: str) -> Dict[str, Any]:
    """
    Map a local intent label (like 'lights.adjust_more') into the router intent schema.
    For now we mostly set type and basic delta hints; brightness parsing stays with GPT or router defaults.
    """
    base = _default_intent()

    if local_intent.startswith("lights.on"):
        base["type"] = "lights.on"
    elif local_intent.startswith("lights.off"):
        base["type"] = "lights.off"
    elif local_intent == "lights.adjust_more":
        base["type"] = "lights.adjust"
        base["delta"] = +100
    elif local_intent == "lights.adjust_less":
        base["type"] = "lights.adjust"
        base["delta"] = -100
    elif local_intent == "lights.set_absolute":
        base["type"] = "lights.adjust"
        # brightness will be handled by GPT or router if not parsed
    elif local_intent == "lights.adjust_percent_delta":
        base["type"] = "lights.adjust"
        # GPT/local numeric parser can refine later
    elif local_intent.startswith("comfort.too_hot"):
        base["type"] = "comfort.too_hot"
    elif local_intent.startswith("comfort.too_cold"):
        base["type"] = "comfort.too_cold"
    elif local_intent.startswith("comfort.set_temp"):
        base["type"] = "comfort.set_temp"
    elif local_intent.startswith("plugs.on"):
        base["type"] = "plugs.on"
    elif local_intent.startswith("plugs.off"):
        base["type"] = "plugs.off"
    elif local_intent.startswith("chat.greeting") or local_intent.startswith("chat.thanks"):
        base["type"] = "chat"
    elif local_intent.startswith("meta.help"):
        base["type"] = "chat"  # router can just answer help via brain.chat
    else:
        base["type"] = "unknown"

    return base


def extract_intent(text: str) -> Dict[str, Any]:
    """
    Main NLU entry point.

    Strategy:
    1. Ask Local NLU (nlu_behemoth_examples) for a label + score.
    2. If score is high enough, map it into the router schema and return.
    3. If score is low or it's clearly small-talk, ask GPT for a structured intent.
    """
    # 1) Local classification
    local_intent, score = local_classify(text)

    # Thresholds are just starting points; tune them based on behaviour.
    HIGH_CONFIDENCE = 0.82
    LOW_CONFIDENCE = 0.55

    if local_intent and score >= HIGH_CONFIDENCE:
        # Strong local hit
        return _map_local_intent_to_schema(local_intent)

    # If we have *some* idea but not strong, you could still use the type hint
    # and then let GPT refine brightness/delta later. For now we go:
    if local_intent and LOW_CONFIDENCE <= score < HIGH_CONFIDENCE:
        # Use local as a hint, but still ask GPT for full structure
        gpt_result = gpt_nlu.gpt_extract_intent(text)
        if gpt_result.get("type") in (None, "", "unknown"):
            # fallback: at least keep the local type hint
            return _map_local_intent_to_schema(local_intent)
        return gpt_result

    # No good local hit → GPT full control
    gpt_result = gpt_nlu.gpt_extract_intent(text)
    if not gpt_result or not gpt_result.get("type"):
        return _default_intent()
    return gpt_result

"""
Intent schema for AshPi.

All NLU engines (local or GPT) should return dictionaries
created through make_intent() so the router can treat them
all the same way.
"""

from typing import Optional, Dict, Any


def make_intent(
    type: str,
    group: Optional[str] = None,
    brightness: Optional[int] = None,
    delta: Optional[int] = None,
    mode: Optional[str] = None,
) -> Dict[str, Any]:
    """
    Standardized intent structure used by AshPi.

    Fields:
      - type: one of
          "lights.on"
          "lights.off"
          "lights.adjust"
          "comfort.cold"
          "chat"
          "unknown"
      - group: logical group name, e.g. "captain_america", "thor", "sky"
      - brightness: direct absolute brightness (0–1000) if applicable
      - delta: relative change in brightness (+/-)
      - mode: optional modifier, e.g. "max", "min"
    """
    return {
        "type": type,
        "group": group,
        "brightness": brightness,
        "delta": delta,
        "mode": mode,
    }

MEMORY_DIR = os.path.expanduser("memory/users")

def _path(user_id):
    return os.path.join(MEMORY_DIR, f"{user_id}.json")

def _ensure_dir():
    os.makedirs(MEMORY_DIR, exist_ok=True)

def load(user_id):
    _ensure_dir()
    path = _path(user_id)
    if not os.path.exists(path):
        return {}
    with open(path, "r") as f:
        return json.load(f)

def save(user_id, data):
    _ensure_dir()
    path = _path(user_id)
    with open(path, "w") as f:
        json.dump(data, f, indent=2)

def get(user_id, key, default=None):
    data = load(user_id)
    return data.get(key, default)

def set(user_id, key, value):
    data = load(user_id)
    data[key] = value
    save(user_id, data)

GROUP_ALIASES = {

def detect_group(text: str):
    t = text.lower()
    for alias, real in GROUP_ALIASES.items():
        if alias in t:

    return None

def detect_max(text: str):
    words = [
        "max",
        "full",
        "full blast",
        "maximum",
        "all the way",
        "super bright",
        "bright as fuck",
        "bright af",
        "blinding",
        "sun mode",
        "bro bright",
    ]
    t = text.lower()
    return any(w in t for w in words)

def detect_adjust(text: str):
    t = text.lower()
    if any(w in t for w in ["darker", "dim", "less", "lower"]):
        return -100
    if any(w in t for w in ["brighter", "lighter", "more", "higher"]):
        return +100
    return None

def detect_on_off(text: str):
    t = text.lower()
    if "turn on" in t or "switch on" in t:
        return "lights.on"
    if "turn off" in t or "switch off" in t:
        return "lights.off"
    return None

def detect_cold(text: str):
    return any(w in text.lower() for w in ["cold", "freezing", "chilly"])

NLU_DATA_FILE = os.path.join(
    os.path.dirname(__file__),
    "nlu_lights_comfort_eg.txt",   # <— your file
)

_EXAMPLES: Dict[str, List[str]] = {}
_LOADED = False

def _load_examples() -> None:
    """Load intent<TAB>utterance lines from the lights+comfort examples file."""
    global _LOADED, _EXAMPLES
    if _LOADED:
        return

    if not os.path.exists(NLU_DATA_FILE):
        # If the file isn't there, just skip; local NLU will still work with
        # the hard-coded rules and GPT via nlu_manager.
        _LOADED = True
        _EXAMPLES = {}
        return

    examples: Dict[str, List[str]] = {}
    with open(NLU_DATA_FILE, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if not line or line.startswith("#"):
                continue
            if "\t" not in line:
                continue
            intent, utterance = line.split("\t", 1)
            intent = intent.strip()
            utt = utterance.strip().lower()
            examples.setdefault(intent, []).append(utt)

    _EXAMPLES = examples
    _LOADED = True

def _classify_from_examples(text: str) -> Tuple[Optional[str], float]:
    """
    Fuzzy match the text against all lights/comfort examples.

    Returns:
        (best_intent_label, score) or (None, 0.0)
    """
    _load_examples()
    if not _EXAMPLES:
        return None, 0.0

    text_norm = text.strip().lower()
    if not text_norm:
        return None, 0.0

    best_intent: Optional[str] = None
    best_score: float = 0.0

    for intent, utterances in _EXAMPLES.items():
        for utt in utterances:
            score = difflib.SequenceMatcher(a=text_norm, b=utt).ratio()
            if score > best_score:
                best_score = score
                best_intent = intent

    return best_intent, best_score

    if label.startswith("comfort.too_cold") \

       or label.startswith("comfort.ack"):
        return make_intent("comfort.cold")

    # Any "chat-like" labels (if they exist in the file)
    if label.startswith("chat.") or label.startswith("meta.help"):
        return make_intent("chat")

    # Anything else → let GPT handle it via nlu_manager
    return make_intent("unknown")

def local_extract_intent(text: str):

        type=intent_type,

        delta=data.get("delta"),

"""
Intent schema for AshPi.

All NLU engines (local or GPT) should return dictionaries
created through make_intent() so the router can treat them
all the same way.
"""

from typing import Optional, Dict, Any


def make_intent(
    type: str,
    group: Optional[str] = None,
    brightness: Optional[int] = None,
    delta: Optional[int] = None,
    mode: Optional[str] = None,
) -> Dict[str, Any]:
    """
    Standardized intent structure used by AshPi.

    Fields:
      - type: one of
          "lights.on"
          "lights.off"
          "lights.adjust"
          "comfort.cold"
          "chat"
          "unknown"
      - group: logical group name, e.g. "captain_america", "thor", "sky"
      - brightness: direct absolute brightness (0–1000) if applicable
      - delta: relative change in brightness (+/-)
      - mode: optional modifier, e.g. "max", "min"
    """
    return {
        "type": type,
        "group": group,
        "brightness": brightness,
        "delta": delta,
        "mode": mode,
    }

def start_cold_session(user_id: str, current_temp: float, preferred_temp: float):
    """
    Start a 'I'm cold' comfort session.

    current_temp   = what the thermostat says now
    preferred_temp = what this user usually likes (from memory or default)
    """

    proposed = preferred_temp

        "preferred": float(preferred_temp),
        "proposed": float(proposed),
        "updated": time.time(),
    }

    user_id = str(user_id)

    user_id = str(user_id)
    s = _sessions.get(user_id)
    if not s:
        return False
    if (time.time() - s["updated"]) > TIMEOUT:
        end_session(user_id)
        return False
    return True


def process_reply(user_id: str, phrase: str):
    """
    Handle follow-ups like:
      - "yes", "sure", "do it"
      - "a little", "just a bit"
      - "one more", "more"
      - "no", "nah", "leave it"

    Returns: (target_temp or None, message)
    If target_temp is not None, caller should actually set the heat.
    """
    user_id = str(user_id)
    if not has_active_session(user_id):
        return None, "No active heat adjustment session."

    s = _sessions[user_id]
    phrase = phrase.lower().strip()
    s["updated"] = time.time()

    # Confirm: set to proposed
    if phrase in ["yes", "sure", "do it", "ok", "okay"]:
        temp = s["proposed"]
        end_session(user_id)
        return temp, f"Turning heat up to {temp}°C."

    # Small nudge from current
    if phrase in ["a little", "just a bit", "a bit"]:
        s["proposed"] = s["current"] + TEMP_STEP
        return s["proposed"], f"Okay, nudging it up to {s['proposed']}°C."

    # More again
    if phrase in ["one more", "more", "bit more"]:
        s["proposed"] += TEMP_STEP
        return s["proposed"], f"Alright, adjusting heat to {s['proposed']}°C."

    # Cancel
    if phrase in ["no", "nah", "leave it", "cancel"]:
        end_session(user_id)
        return None, "Got it — leaving the heat as it is."

    return None, "Not sure what you mean about the heat."

    """Handle /start command."""
    user = update.effective_user

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Pipe every text message into AshPi brain."""
    if not update.message or not update.message.text:
        return

    chat_id = update.effective_chat.id
    text = update.message.text

    logger.info("Telegram message from %s: %s", chat_id, text)

    try:
        # Use chat_id as user_id so each chat has its own brain + sessions
        reply = handle_text(str(chat_id), text)
    except Exception as e:

        reply = "Something blew up in my brain while handling that one, try again in a sec."

    # Send reply back to Telegram
    await update.message.reply_text(reply)

    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
