# Track last home-control action per user
_last_action: Dict[str, Dict[str, object]] = {}


def _set_last_action(user_id: str, kind: str, data: Dict[str, object]):
    _last_action[str(user_id)] = {"kind": kind, "data": data, "ts": time()}


def _get_last_action(user_id: str):
    return _last_action.get(str(user_id))

Step 2 – Add a “perfect/thanks” smart reply

Inside handle_text, after the comfort session block and before # 4) Lights: ON, drop this:

    # 3b) Acks like "perfect", "thanks", etc. right after a home action
    low = text_stripped.lower()
    ack_phrases = {"perfect", "nice", "that’s good", "thats good", "much better", "better", "thanks", "thank you"}

    if low in ack_phrases:
        last = _get_last_action(user_id)
        if last and (time() - last.get("ts", 0) < 20):
            if last["kind"] == "lights":
                data = last["data"]
                group = str(data.get("group", "the lights")).replace("_", " ")
                level = int(data.get("level", 0))
                percent = round(level / 10)
                return f"Got you. Leaving {group} at {percent}%."
            if last["kind"] == "comfort":
                temp = last["data"].get("temp")
                if temp is not None:
                    return f"Sweet. I’ll keep it around {temp}°C."
                return "Got you. I’ll keep it there."
        # No recent home action → just chat normally
        # (falls through to the rest of the router)
