def find_devices(
    backend: str | None = None,
    category: str | None = None,
    room: str | None = None,
    role: str | None = None,
) -> dict:
    """
    Flexible filter for devices.

    Example:
        find_devices(room="bedroom", role="bed_lamp")
        find_devices(category="bulb", room="bedroom")
        find_devices(backend="tuya", category="plug")

    results = {}
    for name, info in DEVICES.items():
        if backend is not None and info.get("backend") != backend:
            continue
        if category is not None and info.get("category") != category:
            continue
        if room is not None and info.get("room") != room:
            continue
        if role is not None and info.get("role") != role:
            continue
        results[name] = info
    return results

    If room is provided, restrict to that room.
    """

# device_manager.py
# Centralized device lookup + filtering logic for AshPi.
# All other modules (router, intents, scenes, automations) should use THIS,
# instead of reading ashpi_devices.DEVICES directly.

import ashpi_devices


def get_device(name: str) -> dict | None:
    """Return a device by its key: 'bed_lamp', 'thor_bulb', etc."""
    return ashpi_devices.DEVICES.get(name)


def find_devices(
    backend: str | None = None,
    category: str | None = None,
    room: str | None = None,
    role: str | None = None,
):
    """
    Flexible search over the device registry.
    Filter by any combination of:
        backend, category, room, role.
    Returns a dict of matching devices.
    """
    result = {}
    for dev_name, dev_info in ashpi_devices.DEVICES.items():
        if backend is not None and dev_info.get("backend") != backend:
            continue
        if category is not None and dev_info.get("category") != category:
            continue
        if room is not None and dev_info.get("room") != room:
            continue
        if role is not None and dev_info.get("role") != role:
            continue
        result[dev_name] = dev_info
    return result


def devices_in_room(room: str):
    """Return all devices located in a given room."""
    return find_devices(room=room)


def devices_with_role(role: str, room: str | None = None):
    """
    Return all devices that have a specific role.
    If room is provided, restrict results to that room.
    """
    return find_devices(role=role, room=room)

"""
Intent schema for AshPi.

All NLU engines (local or GPT) should return dictionaries
created through make_intent() so the router can treat them
all the same way.
"""

from typing import Optional, Dict, Any


def make_intent(
    type: str,
    group: Optional[str] = None,
    brightness: Optional[int] = None,
    delta: Optional[int] = None,
    mode: Optional[str] = None,
) -> Dict[str, Any]:
    """
    Standardized intent structure used by AshPi.

    Fields:
      - type: one of
          "lights.on"
          "lights.off"
          "lights.adjust"
          "comfort.cold"
          "chat"
          "unknown"
      - group: logical group name, e.g. "captain_america", "thor", "sky"
      - brightness: direct absolute brightness (0–1000) if applicable
      - delta: relative change in brightness (+/-)
      - mode: optional modifier, e.g. "max", "min"
    """
    return {
        "type": type,
        "group": group,
        "brightness": brightness,
        "delta": delta,
        "mode": mode,
    }

def _ensure_bulb(name: str):
    if name not in DEVICES:
        raise ValueError(f"Unknown device: {name}")
    dev_type = DEVICES[name].get("type")
    allowed_bulb_types = {"bulb", "tuya_bulb", "light", "tuya_light"}
    if dev_type not in allowed_bulb_types:
        raise ValueError(f"Device {name} is not a bulb (type={dev_type})")


def _get_bulb_ids(target: Union[str, List[str]]):
    names = _normalize_names(target)
    bulbs = []
    for n in names:
        _ensure_bulb(n)
        bulbs.append((n, DEVICES[n]["id"]))
    return bulbs


def _clamp(value: int, min_v: int = 0, max_v: int = 1000) -> int:
    return max(min_v, min(max_v, int(value)))


# ---------- Core actions ----------

def set_brightness(target: Union[str, List[str]], value: int) -> int:
    """
    Set brightness (0–1000) for one or many Tuya bulbs.
    Also turns the LED ON if value > 0, OFF if value == 0.
    """
    bulbs = _get_bulb_ids(target)
    value = _clamp(value)

    for name, dev_id in bulbs:
        BRIGHTNESS_STATE[name] = value

        commands = {
            "commands": [
                {"code": "bright_value_v2", "value": value},
                {"code": "switch_led", "value": value > 0},
            ]
        }
        res = openapi.post(f"/v1.0/devices/{dev_id}/commands", commands)
        logger.info("set_brightness %s -> %s | %s", name, value, res)

    return value


def step_brightness(target: Union[str, List[str]], delta: int = DEFAULT_STEP) -> None:
    """
    Increase or decrease brightness by delta for one or many bulbs.
    Positive delta = brighter, negative delta = dimmer.
    """
    bulbs = _get_bulb_ids(target)

    for name, _ in bulbs:
        current = BRIGHTNESS_STATE.get(name, DEFAULT_BRIGHTNESS)
        new_value = _clamp(current + delta)
        set_brightness(name, new_value)

    """
    Turn bulb(s) on. If value is None, use cached brightness or DEFAULT_BRIGHTNESS.

    # If a single bulb and value is None, use its own cache
    if value is None and isinstance(target, str):
        value = BRIGHTNESS_STATE.get(target, DEFAULT_BRIGHTNESS)

    # If still None (e.g. list), fall back to default
    if value is None:
        value = DEFAULT_BRIGHTNESS

import logging
from typing import List, Union

from tuya_iot import TuyaOpenAPI, TUYA_LOGGER
from env import ENDPOINT, ACCESS_ID, ACCESS_KEY, USERNAME, PASSWORD
from ashpi.devices.ashpi_devices import DEVICES

# ---------- Logging ----------
logging.basicConfig(
    format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
    level=logging.INFO,
)
TUYA_LOGGER.setLevel(logging.INFO)
logger = logging.getLogger("ashpi_lights_tuya")

# ---------- Tuya client ----------
openapi = TuyaOpenAPI(ENDPOINT, ACCESS_ID, ACCESS_KEY)
openapi.connect(USERNAME, PASSWORD, "1", "smartlife")

# ---------- Local brightness cache ----------
# name -> 0–1000
BRIGHTNESS_STATE = {}

DEFAULT_STEP = 100
DEFAULT_BRIGHTNESS = 500  # used if we don't know current brightness yet


def _normalize_names(target: Union[str, List[str]]) -> List[str]:
    if isinstance(target, str):
        return [target]
    return list(target)


def _ensure_bulb(name: str):
    if name not in DEVICES:
        raise ValueError(f"Unknown device: {name}")
    dev_type = DEVICES[name].get("type")
    allowed_bulb_types = {"bulb", "tuya_bulb", "light", "tuya_light", "A19_800lm"}
    if dev_type not in allowed_bulb_types:
        raise ValueError(f"Device {name} is not a bulb (type={dev_type})")


def _get_bulb_ids(target: Union[str, List[str]]):
    names = _normalize_names(target)
    bulbs = []
    for n in names:
        _ensure_bulb(n)
        bulbs.append((n, DEVICES[n]["id"]))
    return bulbs


def _clamp(value: int, min_v: int = 0, max_v: int = 1000) -> int:
    return max(min_v, min(max_v, int(value)))


# ---------- Core actions ----------

def set_brightness(target: Union[str, List[str]], value: int) -> int:
    """
    Set brightness (0–1000) for one or many Tuya bulbs.
    Also turns the LED ON if value > 0, OFF if value == 0.
    """
    bulbs = _get_bulb_ids(target)
    value = _clamp(value)

    for name, dev_id in bulbs:
        BRIGHTNESS_STATE[name] = value

        commands = {
            "commands": [
                {"code": "bright_value_v2", "value": value},
                {"code": "switch_led", "value": value > 0},
            ]
        }
        res = openapi.post(f"/v1.0/devices/{dev_id}/commands", commands)
        logger.info("set_brightness %s -> %s | %s", name, value, res)

    return value


def step_brightness(target: Union[str, List[str]], delta: int = DEFAULT_STEP) -> None:
    """
    Increase or decrease brightness by delta for one or many bulbs.
    Positive delta = brighter, negative delta = dimmer.
    """
    bulbs = _get_bulb_ids(target)

    for name, _ in bulbs:
        current = BRIGHTNESS_STATE.get(name, DEFAULT_BRIGHTNESS)
        new_value = _clamp(current + delta)
        set_brightness(name, new_value)


def turn_on(target: Union[str, List[str]], value: int | None = None) -> None:
    """
    Turn bulb(s) on. If value is None, use cached brightness or DEFAULT_BRIGHTNESS.
    """
    bulbs = _get_bulb_ids(target)

    # If a single bulb and value is None, use its own cache
    if value is None and isinstance(target, str):
        value = BRIGHTNESS_STATE.get(target, DEFAULT_BRIGHTNESS)

    # If still None (e.g. list), fall back to default
    if value is None:
        value = DEFAULT_BRIGHTNESS

    set_brightness(target, value)


def turn_off(target: Union[str, List[str]]) -> None:
    """
    Turn bulb(s) off (switch_led = False, brightness cached as 0).
    """
    bulbs = _get_bulb_ids(target)

    for name, dev_id in bulbs:
        BRIGHTNESS_STATE[name] = 0
        commands = {
            "commands": [
                {"code": "switch_led", "value": False},
            ]
        }
        res = openapi.post(f"/v1.0/devices/{dev_id}/commands", commands)
        logger.info("turn_off %s | %s", name, res)


def get_cached_brightness(name: str) -> int:
    """
    Return last cached brightness for a bulb, or DEFAULT_BRIGHTNESS if unknown.
    (We can later upgrade this to query Tuya live if we want.)
    """
    _ensure_bulb(name)
    return BRIGHTNESS_STATE.get(name, DEFAULT_BRIGHTNESS)

# device_control.py
#
# Bridge between AshPi brain (router) and real Tuya hardware.
# Uses GROUPS to map logical groups -> physical device keys in DEVICES,
# then delegates to ashpi.lights_tuya for actual control.

from ashpi.devices.ashpi_devices import DEVICES
from ashpi.lights_tuya import set_brightness as tuya_set_brightness


# --- Logical group → physical devices mapping --- #

GROUPS = {
    "captain_america": ["captain_america_bulb"],
    "thor": ["thor_bulb"],
    "sky": ["sky_bulb"],
    # later: "bed_lamp": ["bed_lamp"],
}


def set_group_brightness(group: str, level: int):
    """
    group: e.g. 'captain_america'
    level: 0–1000 internal AshPi brightness scale.
    """
    device_keys = GROUPS.get(group)
    if not device_keys:
        print(f"[AshPi WARNING] No devices configured for group '{group}'")
        return

    # Clamp level
    try:
        level = int(level)
    except (TypeError, ValueError):
        level = 0
    level = max(0, min(1000, level))

    # Ensure devices exist
    missing = [k for k in device_keys if k not in DEVICES]
    if missing:
        print(f"[AshPi WARNING] Missing device definitions for: {missing}")
        device_keys = [k for k in device_keys if k in DEVICES]
        if not device_keys:
            return

    print(f"[AshPi INFO] Setting group '{group}' devices {device_keys} to {level}/1000")

from typing import Dict

from ashpi.device_control import set_group_brightness
from .ash_core import AshBrain
from . import comfort_sessions, memory
from .nlu_manager import extract_intent

# In a long-running process (like Telegram bot), we can keep brains in memory:
_brains: Dict[str, AshBrain] = {}

# Simple in-memory lighting sessions:
# _light_sessions[user_id] = {"group": str, "level": int}
_light_sessions: Dict[str, Dict[str, object]] = {}


def get_brain(user_id: str) -> AshBrain:
    user_id = str(user_id)
    if user_id not in _brains:
        _brains[user_id] = AshBrain(user_id)
    return _brains[user_id]


def _get_light_session(user_id: str):
    return _light_sessions.get(str(user_id))


def _start_light_session(user_id: str, group: str, level: int):
    _light_sessions[str(user_id)] = {"group": group, "level": level}


def _update_light_level(user_id: str, level: int):
    if str(user_id) in _light_sessions:
        _light_sessions[str(user_id)]["level"] = level


def _end_light_session(user_id: str):
    _light_sessions.pop(str(user_id), None)


def _clamp_level(level: int) -> int:
    return max(0, min(1000, int(level)))


# --- main entrypoint --- #

def handle_text(user_id: str, text: str) -> str:
    """
    Main entrypoint for AshPi brain.

    Now uses NLU:
      - nlu_manager.extract_intent(text) → intent dict:
        {
          "type": "lights.on" | "lights.off" | "lights.adjust" | "comfort.cold" | "chat" | "unknown",
          "group": "captain_america" | "thor" | "sky" | None,
          "brightness": 0–1000 or None,
          "delta": +/- int or None,
          "mode": "max" | "min" | None,
        }
    """

    user_id = str(user_id)
    text_stripped = text.strip()
    brain = get_brain(user_id)

    # 1) If there's an active HEAT session, let that logic handle yes/no/a bit more/etc.
    if comfort_sessions.has_active_session(user_id):
        temp, msg = comfort_sessions.process_reply(user_id, text_stripped)
        # Later: if temp is not None, call real Nest API here.
        return msg

    # 2) Run NLU on every message
    intent = extract_intent(text_stripped)
    itype = intent["type"]
    group = intent.get("group")
    brightness = intent.get("brightness")
    delta = intent.get("delta")
    mode = intent.get("mode")

    # 3) Comfort / heat starter ("I'm cold", etc.)
    if itype == "comfort.cold":
        # In the future: get real temp from Nest
        current_temp = memory.get(user_id, "last_known_temp", 19.0)
        preferred_temp = memory.get(user_id, "preferred_temp", 21.0)

        if current_temp < preferred_temp - 0.5:
            comfort_sessions.start_cold_session(user_id, current_temp, preferred_temp)
            return (
                f"It's {current_temp}°C in here, and you usually like it around "
                f"{preferred_temp}°. Want me to warm it up?"
            )
        else:
            return "It’s a little cool but still close to what you usually like."

    # 4) Lights: ON
    if itype == "lights.on":
        # If no group was detected, just fall back to chat for now
        if not group:
            return brain.chat(text_stripped)

        # If no brightness specified, default to 600 (60%)
        level = 600 if brightness is None else _clamp_level(brightness)

        set_group_brightness(group, level)
        _start_light_session(user_id, group, level)

        nice_name = group.replace("_", " ")
        if mode == "max":
            return f"Turning on {nice_name} to maximum."
        return f"Turning on {nice_name}."

    # 5) Lights: OFF
    if itype == "lights.off":
        sess = _get_light_session(user_id)

        # Prefer explicit group from intent, else use the active session's group
        target_group = group or (sess["group"] if sess else None)

        if not target_group:
            # No idea which lights to turn off → treat as chat
            return brain.chat(text_stripped)

        set_group_brightness(target_group, 0)
        _end_light_session(user_id)

        nice_name = target_group.replace("_", " ")
        return f"Turning off {nice_name}."

    # 6) Lights: ADJUST (darker / brighter / max / etc.)
    if itype == "lights.adjust":
        sess = _get_light_session(user_id)

        # Determine which group we're adjusting
        if sess and not group:
            target_group = sess["group"]
        else:
            target_group = group

        if not target_group:
            # No active session and no group → we don't know what to adjust
            return brain.chat(text_stripped)

        # Determine starting level
        if sess and sess.get("group") == target_group:
            level = int(sess["level"])
        else:
            # No prior session for this group → start at 600
            level = 600

        # Apply absolute brightness if provided
        if brightness is not None:
            level = _clamp_level(brightness)
        # Else apply delta if provided
        elif delta is not None:
            level = _clamp_level(level + int(delta))

        set_group_brightness(target_group, level)
        _start_light_session(user_id, target_group, level)

        nice_name = target_group.replace("_", " ")
        percent = round(level / 10)
        if level == 0:
            return f"Lights off."
        if mode == "max":
            return f"Setting {nice_name} to maximum brightness."
        return f"Setting {nice_name} to {percent}%."

    # 7) If NLU says "chat" or "unknown" → normal Ash chat
    return brain.chat(text_stripped)
