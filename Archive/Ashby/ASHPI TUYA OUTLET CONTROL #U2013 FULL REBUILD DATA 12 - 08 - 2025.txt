    print("Response:", res)
    return res

import requests
import sys

env = {}
with open("ashpi_nest.env") as f:
    for line in f:
        line = line.strip()
        if not line or line.startswith("#"):
            continue
        if "=" in line:
            key, val = line.split("=", 1)
            val = val.strip().strip('"').strip("'")

NEST_REFRESH_TOKEN = env["NEST_REFRESH_TOKEN"]
NEST_CLIENT_ID = env["NEST_CLIENT_ID"]
NEST_CLIENT_SECRET = env["NEST_CLIENT_SECRET"]
NEST_DEVICE = env["NEST_DEVICE"]

TOKEN_URL = "https://www.googleapis.com/oauth2/v4/token"
EXECUTE_URL = f"https://smartdevicemanagement.googleapis.com/v1/{NEST_DEVICE}:executeCommand"

def get_access_token():
    data = {
        "client_id": NEST_CLIENT_ID,
        "client_secret": NEST_CLIENT_SECRET,
        "refresh_token": NEST_REFRESH_TOKEN,
        "grant_type": "refresh_token",
    }
    r = requests.post(TOKEN_URL, data=data)
    r.raise_for_status()
    body = r.json()
    if "access_token" not in body:
        raise RuntimeError(f"Failed to get access token: {body}")
    return body["access_token"]

def set_heat_celsius(temp_c: float):
    access_token = get_access_token()
    headers = {
        "Authorization": f"Bearer {access_token}",
        "Content-Type": "application/json",
    }
    payload = {
        "command": "sdm.devices.commands.ThermostatTemperatureSetpoint.SetHeat",
        "params": {"heatCelsius": float(temp_c)},
    }
    r = requests.post(EXECUTE_URL, headers=headers, data=json.dumps(payload))
    print("Status:", r.status_code)
    print("Response:", r.text)

if __name__ == "__main__":

        sys.exit(1)

    set_heat_celsius(temp)

Tuya Functions for Bulbs:
-------------------------
Queried via Tuya OpenAPI /v1.0/devices/{id}/functions:

- switch_led       -> Boolean (True/False) = main ON/OFF
- bright_value_v2  -> Integer (0–1000) = brightness level
- work_mode        -> Enum: white/colour/scene/music
- control_data     -> Json: h/s/v, temperature, etc. (for color bulbs)
- scene_data_v2    -> Json: predefined scenes
- countdown_1      -> Integer: timer

For now AshPi uses:
- switch_led (on/off)
- bright_value_v2 (brightness)

ashpi_lights_tuya.py Core Behavior:
-----------------------------------
- BRIGHTNESS_STATE: local cache: { device_name -> 0–1000 }
- DEFAULT_STEP = 100
- DEFAULT_BRIGHTNESS = 500

Functions:

1) set_brightness(target, value)
   - target: single device name or list of names
   - clamps value to [0, 1000]
   - sends Tuya command:
       - bright_value_v2 = value
       - switch_led = True if value > 0 else False
   - updates BRIGHTNESS_STATE for each device

2) step_brightness(target, delta)
   - delta > 0 => brighter
   - delta < 0 => dimmer
   - uses BRIGHTNESS_STATE or DEFAULT_BRIGHTNESS if unknown
   - new_value = clamp(current + delta)
   - internally calls set_brightness()

3) turn_on(target, value=None)
   - If value is None and a single target:
       - uses cached brightness or DEFAULT_BRIGHTNESS
   - If value is None and multiple targets:
       - uses DEFAULT_BRIGHTNESS
   - calls set_brightness(target, value)

4) turn_off(target)
   - sets switch_led = False
   - caches brightness as 0

------------------------------------------------------------

1. **Tuya-based devices give AshPi the easiest and strongest foundation**  
   - Simple local control  
   - Fast response  
   - No cloud dependency  
   - Works for lights, plugs, switches, sensors  

2. **Nest integration is significantly more complex**  
   - Must go through Google Cloud  
   - OAuth tokens  
   - Refresh tokens  
   - Not ideal for quick deployment  
   - Treat Nest as an “advanced optional module,” not core.

3. **AshPi should support multiple brands for future business value**  
   - NOMA iQ (Tuya)  
   - Globe Suite (Tuya)  
   - Future Ayla IoT devices (NOMA thermostats)  
   - Future ONVIF/RTSP cameras  

4. **AshPi must be easy for customers**  
   - User sends text: “Set heat to 22”  
   - AshPi handles everything behind the scenes.

5. **LAN-first design is the priority**  
   - Local control = fast + private  
   - Tuya local keys make this possible  
   - Nest cannot do LAN, so it stays cloud-based but optional.

AshPi Architecture Overview
===========================

Goal:
-----
AshPi must support multiple brands (Tuya, Nest, NOMA iQ, Neewer, etc.) and multiple interfaces
(Telegram, Discord, web app, voice, Twitch, etc.) without duplicating logic.

Key Design:
-----------
1. Brand-specific modules (per vendor/backend)
   - brands/tuya/*   -> Uses Tuya OpenAPI
   - brands/nest/*   -> Uses Google SDM / Nest APIs
   - brands/noma_iq/*-> Uses Ayla backend (future)
   - brands/neewer/* -> Neewer GL1 / other lights

2. Category separation (by device type)
   - lights (bulbs, panels, strips)
   - plugs (smart outlets, power bars)
   - climate (thermostats, heaters)
   - other: garage doors, fans, cameras, etc.

3. Brand-agnostic core
   - core/devices.py: registry of all devices with:
       - id (cloud/device id)
       - type (bulb/plug/thermostat/etc.)
       - brand (tuya, nest, noma_iq, neewer, etc.)
       - room (bedroom, living_room, office, etc.)
       - groups (wall_lights, ceiling_lights, all_lights, etc.)
   - core logic maps phrases like:
